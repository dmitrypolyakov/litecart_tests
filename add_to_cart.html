add_to_cart
open /en/
click css=li.product a.link
waitForElementPresent name=quantity
storeEval Math.floor(Math.random()*9 +1) amount
storeElementPresent css=#box-product .campaign-price hasCampaign
if ${hasCampaign}
storeText css=#box-product .campaign-price price
else
storeText css=#box-product span.price price
endIf
storeEval "${price}".substring(1) price
storeEval ${amount}*${price} total
storeText css=h1.title title
storeElementPresent name=options[Size] hasSize
if ${hasSize}
select name=options[Size] label=Small
endIf
type name=quantity ${amount}
click name=add_cart_product
waitForText css=div#cart a.content Cart:
${amount} item(s) - $${total}
click css=div#cart a.link
waitForElementPresent css=div#checkout-cart-wrapper
assertValue name=quantity ${amount}
assertText //div[@id='box-checkout-cart']/div/ul/li/form/div/p/a/strong ${title}
